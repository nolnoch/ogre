117  void levelSetup(int num) {
118  if(num > 50)
119  num = 50;
120  srand(time(0));
121 
122  Ogre::Plane wallTile = Ogre::Plane(Ogre::Vector3::UNIT_X, -PLANE_DIST +1);
123 
124  // Since each mesh starts at the center of the plane, we need to offset it
125  // to the top right corner of the plane and start counting from there.
126  int offset = WALL_SIZE/2 - TILE_WIDTH/2;
127 
128  int x, y, z;
129  x = y = z = 0;
130 
131  std::vector<int> randomnumbers;
132  for(int i = 0; i < 50; i++)
133  randomnumbers.push_back(i);
134 
135  for(int i = 0; i < num; i++) {
136  std::stringstream ss;
137  std::stringstream ssDebug;
138  ss << (i + tileCounter);
139 
140  int rn = std::rand() % randomnumbers.size(); // get random tile in list of unused tiles
141  int tileNum = randomnumbers[rn];
142  randomnumbers.erase(randomnumbers.begin() + rn);
143  ssDebug << tileNum;
144  // std::cout << "Random number1: " + ssDebug.str() << std::endl;
145  ssDebug.str(std::string());
146 
147  int wallTileNum = tileNum % NUM_TILES_WALL; //possible number of tiles per wall.
148  ssDebug << wallTileNum;
149  // std::cout << "Random number: " + ssDebug.str() << std::endl;
150  int row = wallTileNum / NUM_TILES_ROW; //5 is the number of tiles per row.
151  int col = wallTileNum % NUM_TILES_ROW;
152 
153  ssDebug.str(std::string());
154  ssDebug << row;
155  ssDebug << " ";

157 
158  // std::cout << "Row/col " + ssDebug.str() << std::endl;
159 
160  Ogre::SceneNode* node1; //= mSceneMgr->getRootSceneNode()->createChildSceneNode();
161  int xsize = 240;

163  int zsize = 240;
164 
165  // left
166  if(tileNum < 25) {
167  // set up x y z units of 0, 1 or -1, which will be used when we setPosition
168  // set up our WallTileLeft or whatever to point to the right direction.
169  wallTile = Ogre::Plane(Ogre::Vector3::UNIT_X, 1);

193  wallTile = Ogre::Plane(Ogre::Vector3::NEGATIVE_UNIT_X, 1);

207  // Build the entity name based on which tile number it is.
208  std::string str = "tile";

273  void simonSaysAnim() {
274  if(gameDone) {
275  currTile = -2;

304  mSceneMgr->destroyLight(panelLight);
305 
306  // Create a light
307  panelLight = mSceneMgr->createLight("Panel");
308  panelLight->setCastShadows(false);
309  panelLight->setType(Ogre::Light::LT_SPOTLIGHT);
310  int x = tileSceneNodes[currTile]->_getDerivedPosition().x;
311  int y = tileSceneNodes[currTile]->_getDerivedPosition().y;
312  int z = tileSceneNodes[currTile]->_getDerivedPosition().z;
313 
314  if (x < 0)

316  else if (x > 0)
317  x -= 10;
318 
319  if (z < 0)
320  z += 10;
321  else if (z > 0)
322  z -= 10;
323 
324  // std::cout << "x: " << x << " y: " << y << " z: " << z << "\n";
325  panelLight->setDiffuseColour(0.70, 0.50, 0.30);

328  panelLight->setSpotlightFalloff(0);
329  panelLight->setAttenuation(4000, 0.0, 0.0001, 0.0000005);
330  } else {
331  mSceneMgr->destroyLight(panelLight);
332  panelLight = NULL;
333  }
334 
335  // moves on to the next tile.

337  // std::cout << "c: " << currTile << "\n";
338  }
339  }

341  if (tileEntities.size() > 0) {
342  tileEntities[0]->setMaterialName("Examples/Chrome");
343  animDone = true;
